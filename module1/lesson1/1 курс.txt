#Калькулятор
number_1 = int(input('Введите первое число: '))
number_2 = int(input('Введите второе число: '))

print(number_1, '+', number_2, '=', number_1 + number_2)
print(str(number_1) + '-' + str(number_2) + '=' + str(number_1 - number_2))

print(number_1, '*' , number_2, '=', number_1 * number_2)
print(str(number_1) + '/' + str(number_2) + '=' + str(number_1 / number_2))
